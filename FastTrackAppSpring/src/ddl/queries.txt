CREATE  TABLE `springdb`.`tbl_user_detail` (

  `ID` INT NOT NULL ,

  `USERNAME` VARCHAR(45) NULL ,

  `GENDER` VARCHAR(45) NULL ,

  `PHONENO` INT NULL ,

  `PANNO` VARCHAR(45) NULL );

CREATE  TABLE `springdb`.`mv_route` (

  `ID` INT NOT NULL ,

  `SOURCEDEPO` VARCHAR(45) NULL ,

  `DESTINATIONDEPO` VARCHAR(45) NULL ,

  PRIMARY KEY (`ID`) );

CREATE  TABLE `springdb`.`TBL_BOOKING_STATUS` (

  `ID` INT NOT NULL ,

  `ROUTE_ID` VARCHAR(45) NULL ,

  `SEATS` VARCHAR(45) NULL ,

  `WAITINGLIST` VARCHAR(45) NULL ,

  `BOOKED` VARCHAR(45) NULL ,

  PRIMARY KEY (`ID`) );

CREATE  TABLE `springdb`.`TBL_TICKETS` (

  `ID` INT NOT NULL ,

  `USER_ID` VARCHAR(45) NULL ,

  `ROUTE_ID` VARCHAR(45) NULL ,

  `DOJ` DATE NULL ,

  `AMOUNT` INT NULL ,

  `STATUS` VARCHAR(45) NULL ,

  `NOOFTICKETS` INT NULL ,

  PRIMARY KEY (`ID`) );
